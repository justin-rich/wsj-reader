def get_source
  5.times do |index|
    begin
      p "Downloading #{self.url} (#{index+1})"
      return `curl --connect-timeout #{self.timeout} -s -b#{Merb.root}/config/wsj/cookies.txt "#{self.url}"`        
    rescue Exception => e
      p "There was an error downloading #{self.url}"
      p e
      p e.backtrace          
    end
  end    
  ''         
end
