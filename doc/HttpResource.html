def download_contents
  5.times do |index|
    begin
      p "Downloading #{self.url} (#{index+1})"
      return `curl -s -b#{Merb.root}/config/wsj/cookies.txt "#{self.url}"`        
    rescue Exception => e
      p "There was an error downloading #{self.url}"
      p e
      p e.backtrace          
    end
  end    
  ''         
end

def login

    File.open("#{Merb.root}/config/wsj/cookies.txt").read

  end

end
